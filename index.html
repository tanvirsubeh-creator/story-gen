async function generateScript() { const genBtn = document.getElementById('genBtn'); const status = document.getElementById('status'); genBtn.disabled = true; status.innerText = 'Claude is writing...'; const prompt = `Return ONLY valid JSON. { "title": "...", "body": "...", "views": "..." } Make it a viral TikTok story. Body must be 300+ words.`; try { const res = await fetch("/.netlify/functions/generate", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ prompt }) }); const data = await res.json(); // Check if the function returned an error before trying to read content if (!res.ok || data.error) { status.innerText = 'Error: ' + (data.error || 'Check Console'); genBtn.disabled = false; return; } const contentText = data.content[0].text; const parsed = JSON.parse(contentText); document.getElementById('title').innerText = parsed.title; document.getElementById('views').innerText = "👁 " + parsed.views; document.getElementById('body').innerText = parsed.body; status.innerText = 'Ready!'; } catch (err) { console.error(err); status.innerText = 'System Error (Check Console)'; } genBtn.disabled = false; }
